#include <stdlib.h>

int main() {
    int req[] = {98, 183, 41, 122, 14, 124, 65, 67};
    int n = 8;
    int head = 53;
    int total = 0;

    for(int i = 0; i < n; i++) {
        total += abs(head - req[i]);
        head = req[i];
    }

    printf("Total Head Movement = %d\n", total);

    return 0;
}
