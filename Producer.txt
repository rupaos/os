#include <stdio.h>
#include <semaphore.h>

sem_t mutex, full, empty;
int x = 0;

void producer() {
    sem_wait(&empty);
    sem_wait(&mutex);

    x++;
    printf("Produced: %d\n", x);

    sem_post(&mutex);
    sem_post(&full);
}

void consumer() {
    sem_wait(&full);
    sem_wait(&mutex);

    printf("Consumed: %d\n", x);
    x--;

    sem_post(&mutex);
    sem_post(&empty);
}

int main() {
    sem_init(&mutex, 0, 1);
    sem_init(&full, 0, 0);
    sem_init(&empty, 0, 1);

    producer();
    consumer();

    return 0;
}
