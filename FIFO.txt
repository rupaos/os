#include <stdio.h>

int main() {
    int pages[] = {1, 2, 3, 1, 4};
    int frame[3] = {-1, -1, -1};
    int k = 0, found;

    for (int i = 0; i < 5; i++) {
        found = 0;

        // check if page already in frame
        for (int j = 0; j < 3; j++) {
            if (frame[j] == pages[i]) {
                found = 1;
                break;
            }
        }

        // if not found → replace
        if (!found) {
            frame[k] = pages[i];
            k = (k + 1) % 3;
        }

        // print frames
        for (int j = 0; j < 3; j++) {
            printf("%d ", frame[j]);
        }
        printf("\n");
    }

    return 0;
}
