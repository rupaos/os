

#include <stdio.h>

int main() {
    int bt[3] = {6, 2, 4};

    for(int i=0; i<3; i++) {
        for(int j=i+1; j<3; j++) {
            if(bt[i] > bt[j]) {
                int temp = bt[i];
                bt[i] = bt[j];
                bt[j] = temp;
            }
        }
    }

    printf("Execution Order:\n");

    for(int i=0; i<3; i++) {
        printf("%d ", bt[i]);
    }

    return 0;
}
